\begin{example}[Raw String]
\begin{lstlisting}[language=intentio]
  "foo"      == r"foo"            # output: foo
  "\"foo\""  == r#""foo""#        # output: "foo"
  "x #\"# y" == r##"x #"# y"##    # output: x #"# y
\end{lstlisting}
\end{example}
